docker-compose.yml
version: '3'

services:
  db:
    container_name: evcomp_mysql
    image: mysql
    environment:
      MYSQL_ROOT_PASSWORD: "evcomp"
      MYSQL_DATABASE: "evcomp"
      MYSQL_USER: "evcomp"
      MYSQL_PASSWORD: "evcomp"
    ports:
      - "3315:3306"

  web:
    build: .
    command: python3 manage.py runserver 0.0.0.0:8000
    volumes:
      - .:/code
    ports:
      - "8005:8000"
    depends_on:
      - db

FROM python:3.5
ENV PYTHONUNBUFFERED 1
RUN mkdir /code

ADD requirements.txt /code/
